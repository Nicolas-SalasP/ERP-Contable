-- 2. Crear tabla de mapeo tributario (Traductor Contabilidad -> SII)
CREATE TABLE sii_mapeo_cuentas (
    id INT(11) AUTO_INCREMENT PRIMARY KEY,
    empresa_id INT(11) NOT NULL,
    codigo_cuenta VARCHAR(20) COLLATE utf8mb4_unicode_ci NOT NULL, -- <-- Solución aquí
    concepto_sii ENUM(
        'INGRESOS_DEL_GIRO',
        'OTROS_INGRESOS',
        'REMUNERACIONES_PAGADAS',
        'HONORARIOS_PAGADOS',
        'ARRIENDOS_PAGADOS',
        'GASTOS_GENERALES'
    ) NOT NULL,
    UNIQUE KEY unique_mapeo (empresa_id, codigo_cuenta),
    CONSTRAINT fk_mapeo_empresa FOREIGN KEY (empresa_id) REFERENCES empresas(id) ON DELETE CASCADE,
    CONSTRAINT fk_mapeo_cuenta FOREIGN KEY (codigo_cuenta) REFERENCES plan_cuentas(codigo) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;
